216  inline void drawOne3DObj_cylinder(
217  double cx, double cy, double cz,
218  double radius, double height,
219  double r, double g, double b, double alp)
220  {
221  const int n = 32; // 分段数，控制圆形的平滑度
222  double rd = (long(cx * cy) % 100) * 0.003 + 0.7;
223 
224  glLineWidth(0.1);
225 
226  // 绘制顶面圆盘
227  glBegin(GL_TRIANGLE_FAN);
228  glColor4f(r, g, b, alp);
229  glNormal3d(0, 0, 1); // 法线朝上
230  glVertex3f(cx, cy, cz+ height/2); // 圆心
231  for (int i = 0; i <= n; ++i) {
232  double theta = 2 * PI * i / n;
233  double x = cx + radius * cos(theta);
234  double y = cy + radius * sin(theta);
235  glVertex3f(x,y,cz + height/2);
236  }
237  glEnd();
238 
239  // 绘制底面圆盘
240  glBegin(GL_TRIANGLE_FAN);
241  glColor4f(r, g, b, alp);
242  glNormal3d(0, 0, -1); // 法线朝下
243  glVertex3f(cx, cy, cz- height/2); // 圆心
244  for (int i = n; i >= 0; --i) { // 反向顶点顺序以保证法线正确
245  double theta = 2 * PI * i / n;
246  double x = cx + radius * cos(theta);
247  double y = cy + radius * sin(theta);
248  glVertex3f(x, y, cz - height/2);
249  }
250  glEnd();
251 
252  // 绘制侧面
253  glBegin(GL_QUAD_STRIP);
254  glColor4f(r * rd, g * rd, b * rd, alp); // 侧面颜色调整
255  for (int i = 0; i <= n; ++i) {
256  double theta = 2 * PI * i / n;
257  double cosTheta = cos(theta);
258  double sinTheta = sin(theta);
259  double x = cx + radius * cosTheta;
260  double y = cy + radius * sinTheta;
261 
262  // 设置法线（沿径向方向）
263  glNormal3d(cosTheta, sinTheta,0 );
264  // 侧面四边形顶点
265  glVertex3f(x, y, cz + height/2); // 顶边点
266  glVertex3f(x, y, cz - height/2); // 底边点
267  }
268  glEnd();
269  }
