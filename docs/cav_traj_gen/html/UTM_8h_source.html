116  static inline void LLtoUTM(double Lat, double Long,
117  double &UTMNorthing, double &UTMEasting)
118  {
119 
120  double a = WGS84_A;
121  double eccSquared = UTM_E2;
122  double k0 = UTM_K0;
123 
124  double LongOrigin;
125  double eccPrimeSquared;
126  double N, T, C, A, M;
127 
128  //Make sure the longitude is between -180.00 .. 179.9
129  double LongTemp = (Long+180)-int((Long+180)/360)*360-180;
130 
131  double LatRad = Lat*RADIANS_PER_DEGREE;
132  double LongRad = LongTemp*RADIANS_PER_DEGREE;
133  double LongOriginRad;
134  int ZoneNumber;
135 
136  ZoneNumber = int((LongTemp + 180)/6) + 1;
137 
138  if( Lat >= 56.0 && Lat < 64.0 && LongTemp >= 3.0 && LongTemp < 12.0 )
139  ZoneNumber = 32;
140 
141  // Special zones for Svalbard
142  if( Lat >= 72.0 && Lat < 84.0 )
143  {
144  if( LongTemp >= 0.0 && LongTemp < 9.0 ) ZoneNumber = 31;
145  else if( LongTemp >= 9.0 && LongTemp < 21.0 ) ZoneNumber = 33;
146  else if( LongTemp >= 21.0 && LongTemp < 33.0 ) ZoneNumber = 35;
147  else if( LongTemp >= 33.0 && LongTemp < 42.0 ) ZoneNumber = 37;
148  }
149  // +3 puts origin in middle of zone
150  LongOrigin = (ZoneNumber - 1)*6 - 180 + 3;
151  LongOriginRad = LongOrigin * RADIANS_PER_DEGREE;
152 
153  //compute the UTM Zone from the latitude and longitude
154  char UTMZone[6];
155  sprintf(UTMZone, "%d%c", ZoneNumber, UTMLetterDesignator(Lat));
156 
157  eccPrimeSquared = (eccSquared)/(1-eccSquared);
158 
159  N = a/sqrt(1-eccSquared*sin(LatRad)*sin(LatRad));
160  T = tan(LatRad)*tan(LatRad);
161  C = eccPrimeSquared*cos(LatRad)*cos(LatRad);
162  A = cos(LatRad)*(LongRad-LongOriginRad);
163 
164  M = a*((1 - eccSquared/4 - 3*eccSquared*eccSquared/64
165  - 5*eccSquared*eccSquared*eccSquared/256) * LatRad
166  - (3*eccSquared/8 + 3*eccSquared*eccSquared/32
167  + 45*eccSquared*eccSquared*eccSquared/1024)*sin(2*LatRad)
168  + (15*eccSquared*eccSquared/256
169  + 45*eccSquared*eccSquared*eccSquared/1024)*sin(4*LatRad)
170  - (35*eccSquared*eccSquared*eccSquared/3072)*sin(6*LatRad));
171 
172  UTMEasting = (double)
173  (k0*N*(A+(1-T+C)*A*A*A/6
174  + (5-18*T+T*T+72*C-58*eccPrimeSquared)*A*A*A*A*A/120)
175  + 500000.0);
176 
177  UTMNorthing = (double)
178  (k0*(M+N*tan(LatRad)
179  *(A*A/2+(5-T+9*C+4*C*C)*A*A*A*A/24
180  + (61-58*T+T*T+600*C-330*eccPrimeSquared)*A*A*A*A*A*A/720)));
181 
182  if(Lat < 0)
183  {
184  //10000000 meter offset for southern hemisphere
185  UTMNorthing += 10000000.0;
186  }
187  }

190  static inline void UTMtoLL(const double UTMNorthing, const double UTMEasting,const int UTMzonenum,
191  double& Lat, double& Long )
192  {
193  double k0 = UTM_K0;
194  double a = WGS84_A;
195  double eccSquared = UTM_E2;
196  double eccPrimeSquared;
197  double e1 = (1-sqrt(1-eccSquared))/(1+sqrt(1-eccSquared));
198  double N1, T1, C1, R1, D, M;
199  double LongOrigin;
200  double mu, phi1Rad;
201  double x, y;
202  int ZoneNumber;
203  char* ZoneLetter;
204 
205  x = UTMEasting - 500000.0; //remove 500,000 meter offset for longitude
206  y = UTMNorthing;
207 
208  char UTMZone[6];
209  sprintf(UTMZone, "%d%c", UTMzonenum, 'N');
210 
211  ZoneNumber = strtoul(UTMZone, &ZoneLetter, 10);
212  if((*ZoneLetter - 'N') < 0)
213  {
214  //remove 10,000,000 meter offset used for southern hemisphere
215  y -= 10000000.0;
216  }
217 
218  //+3 puts origin in middle of zone
219  LongOrigin = (ZoneNumber - 1)*6 - 180 + 3;
220  eccPrimeSquared = (eccSquared)/(1-eccSquared);
221 
222  M = y / k0;
223  mu = M/(a*(1-eccSquared/4-3*eccSquared*eccSquared/64
224  -5*eccSquared*eccSquared*eccSquared/256));
225 
226  phi1Rad = mu + ((3*e1/2-27*e1*e1*e1/32)*sin(2*mu)
227  + (21*e1*e1/16-55*e1*e1*e1*e1/32)*sin(4*mu)
228  + (151*e1*e1*e1/96)*sin(6*mu));
229 
230  N1 = a/sqrt(1-eccSquared*sin(phi1Rad)*sin(phi1Rad));
231  T1 = tan(phi1Rad)*tan(phi1Rad);
232  C1 = eccPrimeSquared*cos(phi1Rad)*cos(phi1Rad);
233  R1 = a*(1-eccSquared)/pow(1-eccSquared*sin(phi1Rad)*sin(phi1Rad), 1.5);
234  D = x/(N1*k0);
235 
236  Lat = phi1Rad - ((N1*tan(phi1Rad)/R1)
237  *(D*D/2
238  -(5+3*T1+10*C1-4*C1*C1-9*eccPrimeSquared)*D*D*D*D/24
239  +(61+90*T1+298*C1+45*T1*T1-252*eccPrimeSquared
240  -3*C1*C1)*D*D*D*D*D*D/720));
241 
242  Lat = Lat * DEGREES_PER_RADIAN;
243 
244  Long = ((D-(1+2*T1+C1)*D*D*D/6
245  +(5-2*C1+28*T1-3*C1*C1+8*eccPrimeSquared+24*T1*T1)
246  *D*D*D*D*D/120)
247  / cos(phi1Rad));
248  Long = LongOrigin + Long * DEGREES_PER_RADIAN;
249 
250  }
